estate_clans =
{
	icon = 11
	
	color = { 000 127 014 }
	
	# Is estate kept in control of province on conquest?
	keep_on_conquest = no
	
	# If true, country will get estate
	trigger = {
		OR = {
			has_government_attribute = enables_clans_estate
		}
	}
	
	# If true, province can be granted to estate
	province_trigger = {
		always = yes
	}	

	# Is used to weigh importance compared to other estates
	start_territory_weight = 0
	
	# Min autonomy in estate provinces
	min_autonomy = 25
	
	# Estate cancels out the following multiplicative local autonomy effects:
	 autonomy_does_not_affect = local_manpower_modifier
	 autonomy_does_not_affect = land_forcelimit_modifier

	country_modifier_happy = {
		shock_damage = 0.10
		global_manpower_modifier = 0.20
	}
	country_modifier_neutral = {
		global_manpower_modifier = 0.20
	}
	country_modifier_angry = {
		global_manpower_modifier = -0.20
		shock_damage = -0.10
	}
	
	# These do not scale, but only applied in provinces owned by the estate
	province_modifier_happy = {
		land_forcelimit = 1
		local_manpower = 1
		local_defensiveness = 0.25
	}
	province_modifier_neutral = {
		land_forcelimit = 0.5
		local_manpower = 1
	}	
	province_modifier_angry = {
		local_unrest = 5
	}		
	
	# Influence modifiers
	base_influence = 20
	
	# Influence modifiers from Events:
	
	influence_modifier = {
		desc = EST_VAL_GHULAMS_REFORM
		trigger = {
			has_reform = foreign_ghulams_reform
		}
		influence = 10
	}
	
	
	# Loyalty modifiers
	
	loyalty_modifier = {
		desc = EST_VAL_GHULAMS_REFORM_LOY
		trigger = {
			has_reform = foreign_ghulams_reform
		}
		loyalty = 10
	}
	
	#Alternative naming of estates

	# Does the AI want to give any extra territory beyond the minimum?

	# For province assignment
	ai_grant_province = {
		factor = 1
	}
	
	# Special interactions
	interaction = {
		name = "RECRUIT_CLAN_LEADERS"
		icon = 7
		
		years_between_use = 10
		
		trigger = {
		}
		
		effect = {
			add_estate_influence_modifier = {
				estate = estate_clans
				desc = EST_VAL_BUY_GHULAMS
				influence = 10
				duration = 7300
			}
			add_estate_loyalty = {
				estate = estate_clans
				loyalty = 1
			}
			add_country_modifier = {
				name = "Clans_Leader"
				duration = 3650
			}
		}
	}
}
