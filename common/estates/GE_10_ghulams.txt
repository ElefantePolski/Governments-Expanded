estate_ghulams =
{
	icon = 11
	
	color = { 000 127 014 }
	
	# Is estate kept in control of province on conquest?
	keep_on_conquest = no
	
	# If true, country will get estate
	trigger = {
		OR = {
			has_government_attribute = enables_ghulams_estate
		}
	}
	
	# If true, province can be granted to estate
	province_trigger = {
		always = yes
	}	

	# Is used to weigh importance compared to other estates
	start_territory_weight = 0
	
	# Min autonomy in estate provinces
	min_autonomy = 25

	 autonomy_does_not_affect = local_production_efficiency
	 autonomy_does_not_affect = local_tax_modifier

	country_modifier_happy = {
		army_tradition = 0.5
		state_maintenance_modifier = -0.20
	}
	country_modifier_neutral = {
		state_maintenance_modifier = -0.20
	}
	country_modifier_angry = {
		state_maintenance_modifier = 0.20
		army_tradition = -0.5
	}
	
	# These do not scale, but only applied in provinces owned by the estate
	province_modifier_happy = {
		local_autonomy = -0.10
		local_development_cost = -0.15
	}
	province_modifier_neutral = {
		local_autonomy = -0.10
	}	
	province_modifier_angry = {
		local_unrest = 5

	# Influence modifiers from Events:

		desc = EST_VAL_GHULAMS_REFORM
		trigger = {
			has_reform = foreign_ghulams_reform
		}

	}

		desc = EST_VAL_ABBAS_REFORM_LOY
		trigger = {
			has_reform = reforms_of_abbas_reform
		}
		loyalty = 10
	}
	
	#Alternative naming of estates

	# For province assignment
	ai_grant_province = {
		factor = 1	
		modifier = {
			factor = 0.5
			development = 20
		}
		modifier = {
			factor = 2
			NOT = { development = 10 }
		}				
		modifier = {
			factor = 0.1
			has_estate = estate_cossacks
			has_terrain = steppe
		}
		modifier = {
			factor = 0.75
			has_owner_culture = yes
		}
		modifier = {
			factor = 0.75
			province_trade_power = 5
		}			
		modifier = {
			factor = 0.75
			province_trade_power = 10
		}
		modifier = {
			factor = 4
			local_autonomy = 25
		}
		modifier = {
			factor = 0.25
			NOT = { local_autonomy = 25 }
		}
	}

		icon = 7

							estate = estate_ghulams

					name = "Expanded_Ghulam_Corps_2"

					duration = 7300

				estate = estate_ghulams
				loyalty = -15

				estate = estate_nobles
				loyalty = -5
			}
		}

		icon = 24

				loyalty = 30
			}

				estate = estate_ghulams
				influence = 50
			}
		}
		
		effect = {
			add_estate_influence_modifier = {
				estate = estate_ghulams
				desc = EST_VAL_REPLACE_GOVERNORS
				influence = 10
				duration = 7300
			}
			add_estate_influence_modifier = {
				estate = estate_nobles

				name = "Replaced_Governors"
